class Solution {
public:
    int hammingWeight(int n) {
        int count = 0;
        string binary = "";
       while(n>0){
         binary = char((n % 2) + '0') + binary;
         n /=2;
       }
       for(int i =0; i<binary.size();i++){
        if(binary[i] == '1'){
            count++;
        }
       } 
       return count;
    }
};
