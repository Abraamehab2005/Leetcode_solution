------------------------------------------------------------>

class Solution {
public:
    int maximumDifference(vector<int>& nums) {
        int minn = nums[0];
        int sum = -1;
        for(int i=0;i<nums.size();i++){
            if(nums[i] > minn){
                sum = max(sum,nums[i] - minn);
            }
            minn = min(nums[i],minn);
        }
        return sum;
    }
};
