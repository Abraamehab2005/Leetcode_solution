class Solution {
public:
    vector<int> constructRectangle(int area) {
        int len , wid;
        for(int w = sqrt(area);w>=1;w--){
            if(area % w == 0){
                wid = w;
                len = area / w;
                break;
            }
        }
        return {len,wid};
    }
};
